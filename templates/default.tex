\documentclass[11pt]{article}

% Set margins
\usepackage[margin=1in]{geometry}

% Use Arial as the main font (make sure Arial is installed)
\usepackage{fontspec}
\setmainfont{Arial}

% Set line spacing to 1.15
\usepackage{setspace}
\setstretch{1.15}

% Adjust list spacing (0.2cm between items)
\usepackage{enumitem}
\setlist{
  itemsep=0.2cm,
  parsep=0pt,
  topsep=0.2cm
}

% Add hyperref for hyperlink support
\usepackage[unicode]{hyperref}

% Define \tightlist to avoid errors with Pandoc-generated lists
\providecommand{\tightlist}{%
  % intentionally left empty
}

\setcounter{secnumdepth}{-\maxdimen}

% Format headings using titlesec and ensure headings move to the next page when isolated
\usepackage{titlesec}
\usepackage{needspace}

% Heading Level 1: center, 14pt
\titleformat{\section}
  {\centering\normalfont\fontsize{14}{16}\bfseries}{\thesection}{1em}{}
\let\oldsection\section
\renewcommand{\section}[1]{\Needspace{5\baselineskip}\oldsection{#1}}

% Heading Level 2: 11pt
\titleformat{\subsection}
  {\normalfont\fontsize{11}{13}\bfseries}{\thesubsection}{1em}{}
\let\oldsubsection\subsection
\renewcommand{\subsection}[1]{\Needspace{4\baselineskip}\oldsubsection{#1}}

% Heading Level 3: 11pt
\titleformat{\subsubsection}
  {\normalfont\fontsize{11}{13}\bfseries}{\thesubsubsection}{1em}{}
\let\oldsubsubsection\subsubsection
\renewcommand{\subsubsection}[1]{\Needspace{3\baselineskip}\oldsubsubsection{#1}}
